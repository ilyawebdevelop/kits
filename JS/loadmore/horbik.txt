// каталог
  function btnMoreCatalogInit() {
    var count = 8;
    var selector = '.sect-catalog .products > .swiper-wrapper > .swiper-slide';
    var btnMore = $('.sect-catalog .btn--more');
    var totalCount = $(selector).length;
    $(selector).each(function(index) {
      $(this).hide();
    });
    btnMore.unbind();
    btnMore.show();
    btnMore.click(function(e) {
      e.preventDefault();
      showMore();
    });
    showMore();
    function showMore() {
      $(selector).each(function(index) {
        if (index < count) {
          $(this).fadeIn();
        }
      });
      if (count >= totalCount) {
        btnMore.hide();
      }
      count += 8;
    }
  }
  btnMoreCatalogInit();
